# MySQL性能优化

# source导入太慢

1. 登录mysql 输入密码
mysql -u root -p
2. 创建数据库
create database dbname;
3. 设置参数
# 关闭二进制日志
set sql_log_bin=off;
# 关闭事务自动提交
set autocommit=0;
4. 使用数据库
use dbname;
5. 开启事务
start transaction;
6. 导入sql
source slq-path.sql;
7. 手动提交事务
commit;
8. 改回配置
set sql_log_bin=on;
set autocommit=1;
